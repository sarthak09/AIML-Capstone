for layer in model2.layers:
    layer.trainable = True
    
x1 = model2(x2)

x = Dropout((0.4))(x1)
x = Conv2D(8, (5,4), name="coords2")(x)
x = Dropout((0.4))(x)
x = Conv2D(4, (4,2), name="coords3")(x)
